-- Utility functions
local function stringToVector3(str)
    local x, y, z = str:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")
    return Vector3.new(tonumber(x), tonumber(y), tonumber(z))
end

local function stringToRotationCFrame(str)
    local rx, ry, rz = str:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")
    return CFrame.Angles(math.rad(tonumber(rx)), math.rad(tonumber(ry)), math.rad(tonumber(rz)))
end

local function UpdateTeamOwner()
    local teamColor = LocalPlayer.Team and LocalPlayer.Team.TeamColor.Name or "No Team"
    local shareBlocks = false
    local settings = LocalPlayer:FindFirstChild("Settings")
    if settings then
        local shareBlocksValue = settings:FindFirstChild("ShareBlocks")
        shareBlocks = shareBlocksValue and shareBlocksValue.Value or false
    end
    local teamOwner = LocalPlayer.Name
    if shareBlocks then
        local success, playerListFrame = pcall(function()
            return LocalPlayer.PlayerGui:WaitForChild("PlayerListGui"):WaitForChild("Frame"):WaitForChild("ScrollingFrame")
        end)
        if success and playerListFrame then
            for _, frame in ipairs(playerListFrame:GetChildren()) do
                local playerNameLabel = frame:FindFirstChild("PlayerName")
                local playerRank = frame:FindFirstChild("PlayerRank")
                if playerNameLabel and playerRank then
                    local playerName = playerNameLabel.Text
                    local player = Players:FindFirstChild(playerName)
                    if player and playerRank.Image == "rbxassetid://1912631373" then
                        local playerTeamColor = player.Team and player.Team.TeamColor.Name or "No Team"
                        if playerTeamColor == teamColor then
                            teamOwner = player.Name
                            break
                        end
                    end
                end
            end
        end
    end
    return teamOwner
end

-- Save-related functions
local function GetBuildData(selectedPlayerName)
    local BlocksFolder = GetFolder(selectedPlayerName)
    if not BlocksFolder then
        WindUI:Notify({
            Title = "Error",
            Content = selectedPlayerName .. "'s block folder not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return nil
    end
    local selectedPlayer = Players:FindFirstChild(selectedPlayerName)
    if not selectedPlayer or not selectedPlayer.Team then
        WindUI:Notify({
            Title = "Error",
            Content = selectedPlayerName .. "'s team not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return nil
    end
    local TeamRef = LPTEAM_POSITION(selectedPlayer.Team.Name)
    if not TeamRef then
        WindUI:Notify({
            Title = "Error",
            Content = selectedPlayerName .. "'s team zone not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return nil
    end
    local TeamCF = TeamRef.CFrame
    local blockData = {}

    for _, v in pairs(BlocksFolder:GetChildren()) do
        if v:IsA("Model") then
            local PPart = v:FindFirstChild("PPart")
            if PPart then
                local showShadow = PPart.CastShadow or true
                local canCollide = PPart.CanCollide
                local anchored = PPart.Anchored
                local rotationCF = TeamCF:ToObjectSpace(PPart.CFrame)
                local rx, ry, rz = rotationCF:ToEulerAnglesXYZ()
                local rotationtoString = string.format("%.3f, %.3f, %.3f", math.deg(rx), math.deg(ry), math.deg(rz))
                local position = TeamCF:PointToObjectSpace(PPart.Position)
                local transparency = v:FindFirstChild("TransparencyModifier") and v.TransparencyModifier.Value or 0
                local color = PPart.Color
                local colortoString = string.format("%.6f, %.6f, %.6f", color.R, color.G, color.B)
                local size = string.find(v.Name, "Block") and PPart.Size or nil
                local blockInfo = {
                    ShowShadow = showShadow,
                    CanCollide = canCollide,
                    Color = colortoString,
                    Anchored = anchored,
                    Rotation = rotationtoString,
                    Position = string.format("%.6f, %.6f, %.6f", position.X, position.Y, position.Z),
                    Transparency = transparency
                }
                if size then
                    blockInfo.Size = string.format("%.6f, %.6f, %.6f", size.X, size.Y, size.Z)
                end
                if not blockData[v.Name] then
                    blockData[v.Name] = {}
                end
                table.insert(blockData[v.Name], blockInfo)
            end
        end
    end
    return blockData
end

local function ProcessBuildData(selectedPlayerName, fileName)
    local blockData = GetBuildData(selectedPlayerName)
    if not blockData then
        return
    end
    if next(blockData) then
        local jsonData = {}
        for blockName, blocks in pairs(blockData) do
            local Array = {}
            for _, block in pairs(blocks) do
                local blockJson = {
                    ShowShadow = block.ShowShadow,
                    CanCollide = block.CanCollide,
                    Anchored = block.Anchored,
                    Color = block.Color,
                    Rotation = block.Rotation,
                    Position = block.Position,
                    Transparency = block.Transparency
                }
                if block.Size then
                    blockJson.Size = block.Size
                end
                table.insert(Array, blockJson)
            end
            jsonData[blockName] = Array
        end
        local jsonString = HttpService:JSONEncode(jsonData)
        fileName = fileName or ("Lexushub_" .. math.random(1, 99999))
        local filePath = "LxBuildStorage/" .. fileName .. ".BUILD"
        setclipboard(jsonString)
        writefile(filePath, jsonString)
        WindUI:Notify({
            Title = "Success",
            Content = "JSON copied to clipboard and saved to " .. filePath,
            Icon = "check",
            Duration = 5
        })
    else
        WindUI:Notify({
            Title = "No Blocks",
            Content = "No blocks found for " .. selectedPlayerName .. "!",
            Icon = "info",
            Duration = 5
        })
    end
end

-- Load blocks into workspace
local function LoadBlocksToWorkspace(filePathOrData)
    local RF_BuildingTool = GetToolRF("BuildingTool")
    local RF_ScalingTool = GetToolRF("ScalingTool")
    local RF_PaintingTool = GetToolRF("PaintingTool")
    if not (RF_BuildingTool and RF_ScalingTool and RF_PaintingTool) then
        WindUI:Notify({
            Title = "Error",
            Content = "Required tools not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local stopFlag = false
    local blocksFolder = Workspace:FindFirstChild("Blocks")
    local teamZone = Workspace:FindFirstChild(LocalPlayer.TeamColor.Name .. "Zone")
    local collectedParts = {}
    local paintedParts = {}
    local colors = {}
    local teamOwner = UpdateTeamOwner()
    if not (blocksFolder and teamZone) then
        WindUI:Notify({
            Title = "Error",
            Content = "Blocks or TeamZone not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local function CollectParts(data)
        collectedParts = {}
        paintedParts = {}
        colors = {}
        if stopFlag then return end
        if not data then
            WindUI:Notify({
                Title = "Error",
                Content = "No data provided!",
                Icon = "triangle-alert",
                Duration = 5
            })
            stopFlag = true
            return
        end
        local decoded
        if typeof(data) == "string" then
            local success, content = pcall(readfile, data)
            if not success then
                WindUI:Notify({
                    Title = "Error",
                    Content = "Failed to read file: " .. tostring(content),
                    Icon = "triangle-alert",
                    Duration = 5
                })
                stopFlag = true
                return
            end
            local success, result = pcall(HttpService.JSONDecode, HttpService, content)
            if not success then
                WindUI:Notify({
                    Title = "Error",
                    Content = "Invalid JSON data!",
                    Icon = "triangle-alert",
                    Duration = 5
                })
                stopFlag = true
                return
            end
            decoded = result
        else
            decoded = data
        end
        for blockName, blockList in pairs(decoded) do
            if BuildingParts:FindFirstChild(blockName) then
                for _, blockInfo in ipairs(blockList) do
                    local position = blockInfo.Position and stringToVector3(blockInfo.Position) or Vector3new(0, 0, 0)
                    local rotation = blockInfo.Rotation and stringToRotationCFrame(blockInfo.Rotation) or CFramenew()
                    local cframe = teamZone.CFrame * CFramenew(position) * rotation
                    local size = blockInfo.Size and stringToVector3(blockInfo.Size) or BuildingParts[blockName].PPart.Size
                    local color = blockInfo.Color and Color3new(unpack({blockInfo.Color:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")})) or BuildingParts[blockName].PPart.Color
                    insert(collectedParts, {
                        ParentName = blockName,
                        CFrame = cframe,
                        Size = size,
                        Transparency = blockInfo.Transparency or 0,
                        CanCollide = blockInfo.CanCollide or true,
                        ShowShadow = blockInfo.ShowShadow or true
                    })
                    insert(colors, color)
                end
            end
        end
        if #collectedParts == 0 then
            WindUI:Notify({
                Title = "Error",
                Content = "No valid blocks in data!",
                Icon = "triangle-alert",
                Duration = 5
            })
            stopFlag = true
            return
        end
    end
    local function InvokeBuildingTool(partName, dataValue, cframe)
        local success, result = pcall(function()
            RF_BuildingTool:InvokeServer(partName, dataValue, nil, nil, true, cframe, false)
        end)
        if not success then
            warn("BuildingTool failed: " .. tostring(result))
            return false
        end
        return true
    end
    local function InvokeScalingTool(part, size, cframe)
        local success, result = pcall(function()
            RF_ScalingTool:InvokeServer(part, size, cframe)
        end)
        if not success then
            warn("ScalingTool failed: " .. tostring(result))
            return false
        end
        return true
    end
    local function InvokePaintingTool(parts)
        local success, result = pcall(function()
            RF_PaintingTool:InvokeServer(parts)
        end)
        if not success then
            warn("PaintingTool failed: " .. tostring(result))
            return false
        end
        return true
    end
    local function RenameBlock(block, newName)
        block.Name = newName
    end
    local function ProcessPart(partData, index)
        local partName = partData.ParentName
        local dataValue = Data and Data:FindFirstChild(partName) and Data[partName].Value or nil
        if not dataValue then
            warn("No data value for " .. partName)
            WindUI:Notify({
                Title = "Error",
                Content = "Missing data for: " .. partName,
                Icon = "triangle-alert",
                Duration = 5
            })
            return
        end
        local offsetY = 1000 + (index * 1)
        local cframe = partName:lower():find("block") and teamZone.CFrame * CFramenew(0, offsetY, 0) or partData.CFrame
        if not InvokeBuildingTool(partName, dataValue, cframe) then
            WindUI:Notify({
                Title = "Error",
                Content = "Failed to place: " .. partName,
                Icon = "triangle-alert",
                Duration = 5
            })
        end
    end
    local partCounter = 0
    local startTime = tick()
    local function ProcessPartAfterScaling(partData, index)
        partCounter = partCounter + 1
        UpdateProgressParagraph(partCounter, #collectedParts, startTime)
        local partName = partData.ParentName
        local ownerBlocks = blocksFolder:FindFirstChild(teamOwner)
        local block = ownerBlocks and ownerBlocks:WaitForChild(partName, 2)
        if block then
            local newName = partName .. tostring(index)
            RenameBlock(block, newName)
            if InvokeScalingTool(block, partData.Size, partData.CFrame) then
                if colors and #colors >= index then
                    insert(paintedParts, {block, colors[index]})
                else
                    warn("Color index out of bounds for " .. partName)
                end
            else
                WindUI:Notify({
                    Title = "Error",
                    Content = "Failed to scale: " .. partName,
                    Icon = "triangle-alert",
                    Duration = 5
                })
            end
            task.defer(function()
                if block then RenameBlock(block, partName) end
            end)
        else
            warn("Block not found: " .. partName)
            WindUI:Notify({
                Title = "Error",
                Content = "Block not found in Blocks." .. teamOwner,
                Icon = "triangle-alert",
                Duration = 5
            })
        end
    end
    local function ProcessAllParts()
        task.defer(function()
            for i, partData in ipairs(collectedParts) do
                if stopFlag then break end
                task.spawn(function()
                    ProcessPart(partData, i)
                    task.defer(function()
                        task.wait(0.5)
                        ProcessPartAfterScaling(partData, i)
                    end)
                end)
            end
            repeat
                task.wait()
            until #collectedParts == partCounter or stopFlag
            if #collectedParts == partCounter and not stopFlag then
                collectedParts = {}
                WindUI:Notify({
                    Title = "Success",
                    Content = "Build loaded successfully. Wait a few seconds for paint to load or paint may fail for some blocks",
                    Icon = "check",
                    Duration = 5
                })
                task.wait(7)
                if not InvokePaintingTool(paintedParts) then
                    WindUI:Notify({
                        Title = "Warning",
                        Content = "Painting failed for some blocks",
                        Icon = "triangle-alert",
                        Duration = 5
                    })
                end
            else
                WindUI:Notify({
                    Title = "Error",
                    Content = "Incomplete: " .. partCounter .. "/" .. #collectedParts .. " blocks placed",
                    Icon = "triangle-alert",
                    Duration = 5
                })
            end
        end)
    end
    local function StartProcess()
        UpdateProgressParagraph(0, 0, startTime)
        CollectParts(filePathOrData)
        if stopFlag then
            return
        end
        local connection
        connection = cancelButton.MouseButton1Click:Connect(function()
            stopFlag = true
            connection:Disconnect()
            WindUI:Notify({
                Title = "Build Cancelled",
                Content = "Build process stopped",
                Icon = "info",
                Duration = 5
            })
        end)
        ProcessAllParts()
    end
    StartProcess()
end

-- Add at the top with other global variables
local previewHighlightColor = Color3.fromRGB(0, 255, 255) -- Default cyan for the BuildPreview highlight
local usePerBlockHighlight = false -- Toggle between single model highlight (false) and per-block highlight (true)

-- Modified PreviewFile function to highlight the entire BuildPreview model
local function PreviewFile(filePath, scale, team)
    local plot = team and TeamZones[team] or Zone
    if not plot then
        WindUI:Notify({
            Title = "Error",
            Content = "Team plot not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local success, fileContent = pcall(readfile, filePath)
    if not success then
        WindUI:Notify({
            Title = "Error",
            Content = "Failed to read file: " .. tostring(fileContent),
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local decoded
    success, decoded = pcall(HttpService.JSONDecode, HttpService, fileContent)
    if not success then
        WindUI:Notify({
            Title = "Error",
            Content = "Invalid JSON: " .. tostring(decoded),
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    BuildPreview:ClearAllChildren()
    local blockCount = 0
    for blockName, blockList in pairs(decoded) do
        if BuildingParts:FindFirstChild(blockName) then
            for _, block in ipairs(blockList) do
                task.spawn(function()
                    local position = block.Position and stringToVector3(block.Position) or Vector3new(0, 0, 0)
                    local rotation = block.Rotation and stringToRotationCFrame(block.Rotation) or CFramenew()
                    local cframe = plot.CFrame * CFramenew(position * (scale or 1)) * rotation
                    local clonedBlock = Clone(BuildingParts[blockName])
                    SetPrimaryPartCFrame(clonedBlock, cframe)
                    clonedBlock.Health.Value = ""
                    clonedBlock.Parent = BuildPreview
                    local pPart = clonedBlock:FindFirstChild("PPart")
                    if pPart then
                        pPart.Size = block.Size and stringToVector3(block.Size) * (scale or 1) or pPart.Size
                        pPart.Anchored = true
                        pPart.Transparency = block.Transparency or 0.5 -- Slightly transparent for preview
                        pPart.CastShadow = block.ShowShadow or false
                        pPart.CanCollide = block.CanCollide or true
                        if block.Color then
                            local color = Color3new(unpack({block.Color:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")}))
                            for _, descendant in pairs(GetDescendants(clonedBlock)) do
                                if descendant:IsA("BasePart") then
                                    descendant.Color = color
                                end
                            end
                        end
                        -- Per-block highlight if enabled
                        if usePerBlockHighlight then
                            local highlight = Instance.new("Highlight")
                            highlight.Name = "BlockHighlight"
                            highlight.Adornee = pPart
                            highlight.FillTransparency = 1 -- No fill, only outline
                            highlight.OutlineColor = previewHighlightColor
                            highlight.OutlineTransparency = 0 -- Fully opaque outline
                            highlight.Parent = pPart
                        end
                    else
                        warn("No PPart found in cloned block: " .. blockName)
                    end
                    blockCount = blockCount + 1
                    task.wait()
                end)
            end
        end
    end
    -- Add single Highlight to the entire BuildPreview model (default approach)
    if blockCount > 0 and not usePerBlockHighlight then
        local highlight = Instance.new("Highlight")
        highlight.Name = "BuildPreviewHighlight"
        highlight.Adornee = BuildPreview
        highlight.FillTransparency = 1 -- No fill, only outline
        highlight.OutlineColor = previewHighlightColor
        highlight.OutlineTransparency = 0 -- Fully opaque outline
        highlight.Parent = BuildPreview
        -- Debug: Check if all blocks are properly parented
        for _, child in pairs(BuildPreview:GetChildren()) do
            if child:IsA("Model") and not child:FindFirstChild("PPart") then
                warn("Block without PPart detected in BuildPreview: " .. child.Name)
            end
        end
    end
    WindUI:Notify({
        Title = blockCount > 0 and "Success" or "No Blocks",
        Content = blockCount > 0 and ("Previewed " .. blockCount .. " blocks with highlight") or "No valid blocks in file!",
        Icon = blockCount > 0 and "check" or "info",
        Duration = 5
    })
end

-- Modified ClearPreview function to ensure all highlights are removed
local function ClearPreview()
    BuildPreview:ClearAllChildren()
    if PreviewPart then
        PreviewPart:Destroy()
        PreviewPart = nil
    end
    BuildPreview.PrimaryPart = nil
    -- Ensure any existing highlight is removed
    local existingHighlight = BuildPreview:FindFirstChild("BuildPreviewHighlight")
    if existingHighlight then
        existingHighlight:Destroy()

-- Function to load blocks into workspace
local function LoadBlocksToWorkspace(filePathOrData)
    local RF_BuildingTool = GetToolRF("BuildingTool")
    local RF_ScalingTool = GetToolRF("ScalingTool")
    local RF_PaintingTool = GetToolRF("PaintingTool")
    if not (RF_BuildingTool and RF_ScalingTool and RF_PaintingTool) then
        WindUI:Notify({
            Title = "Error",
            Content = "Required tools not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local stopFlag = false
    local blocksFolder = Workspace:FindFirstChild("Blocks")
    local teamZone = Workspace:FindFirstChild(LocalPlayer.TeamColor.Name .. "Zone")
    local collectedParts = {}
    local paintedParts = {}
    local colors = {}
    local teamOwner = UpdateTeamOwner()
    if not (blocksFolder and teamZone) then
        WindUI:Notify({
            Title = "Error",
            Content = "Blocks or TeamZone not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local function CollectParts(data)
        collectedParts = {}
        paintedParts = {}
        colors = {}
        if stopFlag then return end
        if not data then
            WindUI:Notify({
                Title = "Error",
                Content = "No data provided!",
                Icon = "triangle-alert",
                Duration = 5
            })
            stopFlag = true
            return
        end
        local decoded
        if typeof(data) == "string" then
            local success, content = pcall(readfile, data)
            if not success then
                WindUI:Notify({
                    Title = "Error",
                    Content = "Failed to read file: " .. tostring(content),
                    Icon = "triangle-alert",
                    Duration = 5
                })
                stopFlag = true
                return
            end
            local success, result = pcall(HttpService.JSONDecode, HttpService, content)
            if not success then
                WindUI:Notify({
                    Title = "Error",
                    Content = "Invalid JSON data!",
                    Icon = "triangle-alert",
                    Duration = 5
                })
                stopFlag = true
                return
            end
            decoded = result
        else
            decoded = data
        end
        for blockName, blockList in pairs(decoded) do
            if BuildingParts:FindFirstChild(blockName) then
                for _, blockInfo in ipairs(blockList) do
                    local position = blockInfo.Position and stringToVector3(blockInfo.Position) or Vector3new(0, 0, 0)
                    local rotation = blockInfo.Rotation and stringToRotationCFrame(blockInfo.Rotation) or CFramenew()
                    local cframe = teamZone.CFrame * CFramenew(position) * rotation
                    local size = blockInfo.Size and stringToVector3(blockInfo.Size) or BuildingParts[blockName].PPart.Size
                    local color = blockInfo.Color and Color3new(unpack({blockInfo.Color:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")})) or BuildingParts[blockName].PPart.Color
                    insert(collectedParts, {
                        ParentName = blockName,
                        CFrame = cframe,
                        Size = size,
                        Transparency = blockInfo.Transparency or 0,
                        CanCollide = blockInfo.CanCollide or true,
                        ShowShadow = blockInfo.ShowShadow or true
                    })
                    insert(colors, color)
                end
            end
        end
        if #collectedParts == 0 then
            WindUI:Notify({
                Title = "Error",
                Content = "No valid blocks in data!",
                Icon = "triangle-alert",
                Duration = 5
            })
            stopFlag = true
            return
        end
        print("Collected parts: " .. #collectedParts)
    end
    local function InvokeBuildingTool(partName, dataValue, cframe)
        local success, result = pcall(function()
            RF_BuildingTool:InvokeServer(partName, dataValue, nil, nil, true, cframe, false)
        end)
        if not success then
            warn("BuildingTool failed: " .. tostring(result))
            return false
        end
        return true
    end
    local function InvokeScalingTool(part, size, cframe)
        local success, result = pcall(function()
            RF_ScalingTool:InvokeServer(part, size, cframe)
        end)
        if not success then
            warn("ScalingTool failed: " .. tostring(result))
            return false
        end
        return true
    end
    local function InvokePaintingTool(parts)
        local success, result = pcall(function()
            RF_PaintingTool:InvokeServer(parts)
        end)
        if not success then
            warn("PaintingTool failed: " .. tostring(result))
            return false
        end
        return true
    end
    local function RenameBlock(block, newName)
        block.Name = newName
    end
    local function ProcessPart(partData, index)
        local partName = partData.ParentName
        local dataValue = Data and Data:FindFirstChild(partName) and Data[partName].Value or nil
        if not dataValue then
            warn("No data value for " .. partName)
            WindUI:Notify({
                Title = "Error",
                Content = "Missing data for: " .. partName,
                Icon = "triangle-alert",
                Duration = 5
            })
            return
        end
        local offsetY = 1000 + (index * 1)
        local cframe = partName:lower():find("block") and teamZone.CFrame * CFramenew(0, offsetY, 0) or partData.CFrame
        if not InvokeBuildingTool(partName, dataValue, cframe) then
            WindUI:Notify({
                Title = "Error",
                Content = "Failed to place: " .. partName,
                Icon = "triangle-alert",
                Duration = 5
            })
        end
    end
    local partCounter = 0
    local startTime = tick()
    local function ProcessPartAfterScaling(partData, index)
        partCounter = partCounter + 1
        UpdateProgressParagraph(partCounter, #collectedParts, startTime)
        local partName = partData.ParentName
        local ownerBlocks = blocksFolder:FindFirstChild(teamOwner)
        local block = ownerBlocks and ownerBlocks:WaitForChild(partName, 2)
        if block then
            local newName = partName .. tostring(index)
            RenameBlock(block, newName)
            if InvokeScalingTool(block, partData.Size, partData.CFrame) then
                if colors and #colors >= index then
                    insert(paintedParts, {block, colors[index]})
                else
                    warn("Color index out of bounds for " .. partName)
                end
            else
                WindUI:Notify({
                    Title = "Error",
                    Content = "Failed to scale: " .. partName,
                    Icon = "triangle-alert",
                    Duration = 5
                })
            end
            task.defer(function()
                if block then RenameBlock(block, partName) end
            end)
        else
            warn("Block not found: " .. partName)
            WindUI:Notify({
                Title = "Error",
                Content = "Block not found in Blocks." .. teamOwner,
                Icon = "triangle-alert",
                Duration = 5
            })
        end
    end
    local function ProcessAllParts()
        task.defer(function()
            for i, partData in ipairs(collectedParts) do
                if stopFlag then break end
                task.spawn(function()
                    ProcessPart(partData, i)
                    task.defer(function()
                        task.wait(0.5)
                        ProcessPartAfterScaling(partData, i)
                    end)
                end)
            end
            repeat
                task.wait(0.5)
            until #collectedParts == partCounter or stopFlag
            if #collectedParts == partCounter and not stopFlag then
                collectedParts = {}
                WindUI:Notify({
                    Title = "Success",
                    Content = "Build loaded successfully What Few seconds for the paint to Load or Paint May fail For some Blocks",
                    Icon = "check",
                    Duration = 5
                })
                task.wait(5)
                if not InvokePaintingTool(paintedParts) then
                    WindUI:Notify({
                        Title = "Warning",
                        Content = "Painting failed for some blocks",
                        Icon = "triangle-alert",
                        Duration = 5
                    })
                end
            else
                WindUI:Notify({
                    Title = "Error",
                    Content = "Incomplete: " .. partCounter .. "/" .. #collectedParts .. " blocks placed",
                    Icon = "triangle-alert",
                    Duration = 5
                })
            end
            -- Don't clear GUI automatically per user preference
        end)
    end
    local function StartProcess()
        UpdateProgressParagraph(0, 0, startTime)
        CollectParts(filePathOrData)
        if stopFlag then
            return
        end
        -- Connect Cancel button
        local connection
        connection = cancelButton.MouseButton1Click:Connect(function()
            stopFlag = true
            connection:Disconnect()
            WindUI:Notify({
                Title = "Build Cancelled",
                Content = "Build process stopped",
                Icon = "info",
                Duration = 5
            })
            -- Don't clear GUI per user preference
        end)
        ProcessAllParts()
    end
    StartProcess()
end

 pPart.Anchored = true

                    Redthing.FillColor = Color3.fromRGB(100, 0, 0)
	                Redthing.OutlineColor = Color3.fromRGB(0, 0, 0)
                    highlight.OutlineTransparency = 0 -- Fully opaque outline
                    highlight.Parent = workpace.BuildPreview

                    blockCount = blockCount + 1
                    task.wait()
                end)
            end
        end
    end
    WindUI:Notify({
        Title = blockCount > 0 and "Success" or "No Blocks",
        Content = blockCount > 0 and ("Previewed " .. blockCount .. " blocks") or "No valid blocks in file!",
        Icon = blockCount > 0 and "check" or "info",
        Duration = 5
    })
end

-- Constants
local RADIANS_TO_DEGREES = 57.29577951308232
local DEFAULT_BLOCK_SIZE = Vector3.new(2, 2, 2)

-- Helper functions for CFrame and string manipulation
local function GetStringAngles(cframe)
    local x, y, z = cframe:ToEulerAnglesXYZ()
    x = math.floor((x * RADIANS_TO_DEGREES) * 100000 + 0.5) / 100000
    y = math.floor((y * RADIANS_TO_DEGREES) * 100000 + 0.5) / 100000
    z = math.floor((z * RADIANS_TO_DEGREES) * 100000 + 0.5) / 100000
    return string.format("%s,%s,%s", x, y, z)
end

local function StringifyVector(vector)
    return string.gsub(tostring(vector), " ", "")
end

local function AnglesString(angleString)
    local angles = string.split(angleString, ",")
    return CFrame.Angles(math.rad(angles[1]), math.rad(angles[2]), math.rad(angles[3]))
end

-- Encode block data into a JSON-compatible table
local function Encode(blocks, player)
    local jsonTable = {}
    local plot = GetPlot(player)

    for _, block in ipairs(blocks) do
        local blockName = block.Name
        local primaryPart = block:FindFirstChild("PPart")
        if not primaryPart then
            continue
        end

        if not jsonTable[blockName] then
            jsonTable[blockName] = {}
        end

        local relativeCFrame = plot.CFrame:ToObjectSpace(primaryPart.CFrame)
        local buildingParts = game:GetService("ReplicatedStorage").BuildingParts
        local defaultColor = buildingParts[blockName].PPart.Color

        table.insert(jsonTable[blockName], {
            Rotation = GetStringAngles(relativeCFrame),
            Position = StringifyVector(relativeCFrame.Position),
            ShowShadow = primaryPart.CastShadow or true,
            CanCollide = primaryPart.CanCollide or true,
            Anchored = primaryPart.Anchored or true,
            Transparency = primaryPart.Transparency > 0 and primaryPart.Transparency or nil,
            Size = string.find(blockName, "Block") and primaryPart.Size ~= DEFAULT_BLOCK_SIZE and StringifyVector(primaryPart.Size) or nil,
            Color = primaryPart.Color ~= defaultColor and StringifyVector(primaryPart.Color) or nil
        })
    end

    return HttpService:JSONEncode(jsonTable)
end

local function EnsurePreviewPart()
    if not BuildPreview then
        BuildPreview = Instance.new("Model", Workspace)
        BuildPreview.Name = "BuildPreview"
    end
    if not PreviewPart or PreviewPart.Parent ~= BuildPreview then
        PreviewPart = Instance.new("Part", BuildPreview)
        PreviewPart.Name = "PreviewPart"
        PreviewPart.Transparency = 1
        PreviewPart.Anchored = true
        PreviewPart.CanCollide = false
        PreviewPart.CFrame = Zone and Zone.CFrame or CFramenew()

    if not BuildPreview.PrimaryPart then
        BuildPreview.PrimaryPart = PreviewPart
    end
end

local function ApplyPreviewTransform(positionOffset, scaleChange, rotationChange)
    if not Zone then
        WindUI:Notify({
            Title = "Error",
            Content = "Team plot not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    if not BuildPreview or not BuildPreview.Parent then
        WindUI:Notify({
            Title = "Error",
            Content = "BuildPreview model not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    EnsurePreviewPart()
    local blockCount = 0
    for _, child in ipairs(BuildPreview:GetChildren()) do
        if child:IsA("Model") and child:FindFirstChild("PPart") then
            blockCount = blockCount + 1
        end
    end
    if blockCount == 0 and not (positionOffset or scaleChange or rotationChange) then
        WindUI:Notify({
            Title = "Error",
            Content = "No blocks in BuildPreview to adjust!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    if positionOffset then
        local currentCFrame = BuildPreview.PrimaryPart.CFrame
        local relativeOffset = Zone.CFrame:VectorToWorldSpace(positionOffset * movementMultiplier)
        BuildPreview:SetPrimaryPartCFrame(currentCFrame + relativeOffset)
    end
    if rotationChange then
        rotationAngles.X = rotationChange.X or rotationAngles.X
        rotationAngles.Y = rotationChange.Y or rotationAngles.Y
        rotationAngles.Z = rotationChange.Z or rotationAngles.Z
        local rotationCFrame = CFrameAng(
            rad(rotationAngles.X),
            rad(rotationAngles.Y),
            rad(rotationAngles.Z)
        )
        local currentCFrame = BuildPreview.PrimaryPart.CFrame
        local position = currentCFrame.Position
        BuildPreview:SetPrimaryPartCFrame(CFramenew(position) * rotationCFrame)
    end
    if scaleChange and scaleChange ~= 1 then
        sizeScale = math.clamp(sizeScale * scaleChange, 0.1, 10)
        for _, model in ipairs(BuildPreview:GetChildren()) do
            if model:IsA("Model") and model:FindFirstChild("PPart") then
                local pPart = model.PPart
                if not model:FindFirstChild("OriginalSize") then
                    local originalSize = Instance.new("Vector3Value", model)
                    originalSize.Name = "OriginalSize"
                    originalSize.Value = pPart.Size
                end
                local originalSize = model.OriginalSize.Value
                pPart.Size = originalSize * sizeScale
                local relativeCFrame = BuildPreview.PrimaryPart.CFrame:ToObjectSpace(pPart.CFrame)
                pPart.CFrame = BuildPreview.PrimaryPart.CFrame * CFramenew(relativeCFrame.Position * sizeScale) * relativeCFrame:ToOrientation()
            end
        end
    end
    WindUI:Notify({
        Title = "Transformation Applied",
        Content = "Position, rotation, and/or scale updated",
        Icon = "check",
        Duration = 2
    })
end

-- Function to convert model to .Build format with PlasticBlock
local function ConvertModelToBuild(model, saveName)
    if not Zone then
        WindUI:Notify({
            Title = "Error",
            Content = "TeamZone not found! Ensure you're on a team and the zone exists.",
            Icon = "triangle-alert",
            Duration = 5
        })
        return nil
    end
    local blockData = { PlasticBlock = {} }
    local parts = {}
    for _, descendant in pairs(model:GetDescendants()) do
        if descendant:IsA("BasePart") then
            table.insert(parts, descendant)
        end
    end
    if #parts == 0 then
        WindUI:Notify({
            Title = "Error",
            Content = "No valid parts found in model!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return nil
    end
    for _, part in ipairs(parts) do
        local rotationCF = Zone.CFrame:ToObjectSpace(part.CFrame)
        local rx, ry, rz = rotationCF:ToEulerAnglesXYZ()
        local rotationtoString = string.format("%.3f, %.3f, %.3f", math.deg(rx), math.deg(ry), math.deg(rz))
        local position = Zone.CFrame:PointToObjectSpace(part.Position)
        local transparency = part.Transparency
        local color = part.Color
        local colortoString = string.format("%.6f, %.6f, %.6f", color.R, color.G, color.B)
        local size = part.Size
        local blockInfo = {
            ShowShadow = part.CastShadow,
            CanCollide = part.CanCollide,
            Color = colortoString,
            Anchored = part.Anchored,
            Rotation = rotationtoString,
            Position = string.format("%.6f, %.6f, %.6f", position.X, position.Y, position.Z),
            Transparency = transparency,
            Size = string.format("%.6f, %.6f, %.6f", size.X, size.Y, size.Z)
        }
        -- Check for Servo or similar Bool/NumberValues
        for _, child in pairs(part:GetChildren()) do
            if child:IsA("BoolValue") or child:IsA("NumberValue") then
                blockInfo[child.Name] = child.Value
            end
        end
        table.insert(blockData.PlasticBlock, blockInfo)
    end
    if #blockData.PlasticBlock > 0 then
        local jsonString = HttpService:JSONEncode(blockData)
        local filePath = "LxBuildStorage/" .. saveName .. ".BUILD"
        local success, errorMsg = pcall(function()
            writefile(filePath, jsonString) -- Fixed missing parenthesis
        end)
        if success then
            WindUI:Notify({
                Title = "Success",
                Content = "Model saved as " .. filePath,
                Icon = "check",
                Duration = 5
            })
            return filePath
        else
            WindUI:Notify({
                Title = "Error",
                Content = "Failed to save file: " .. tostring(errorMsg),
                Icon = "triangle-alert",
                Duration = 5
            })
            return nil
        end
    else
        WindUI:Notify({
            Title = "Error",
            Content = "No valid parts found in model!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return nil
    end
end

-- Step 1: Function to apply mesh refinements to a build file
local function ApplyMeshRefinement(filePath, meshLevel, transparency)
    local plot = Zone
    if not plot then
        WindUI:Notify({
            Title = "Error",
            Content = "Team plot not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local success, fileContent = pcall(readfile, filePath)
    if not success then
        WindUI:Notify({
            Title = "Error",
            Content = "Failed to read file: " .. tostring(fileContent),
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local decoded
    success, decoded = pcall(HttpService.JSONDecode, HttpService, fileContent)
    if not success then
        WindUI:Notify({
            Title = "Error",
            Content = "Invalid JSON: " .. tostring(decoded),
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    BuildPreview:ClearAllChildren()
    local blockCount = 0
    -- Inverse mesh level: higher mesh level means smaller blocks (less blocky)
    local scaleFactor = 1 / math.max(0.1, meshLevel) -- e.g., meshLevel 2 -> scaleFactor 0.5
    for blockName, blockList in pairs(decoded) do
        if BuildingParts:FindFirstChild(blockName) then
            for _, block in ipairs(blockList) do
                task.spawn(function()
                    -- Parse position and rotation
                    local position = block.Position and stringToVector3(block.Position) or Vector3new(0, 0, 0)
                    local rotation = block.Rotation and stringToRotationCFrame(block.Rotation) or CFramenew()
                    -- Scale position to reduce gaps
                    local scaledPosition = position * scaleFactor
                    local cframe = plot.CFrame * CFramenew(scaledPosition) * rotation
                    -- Clone block and set properties
                    local clonedBlock = Clone(BuildingParts[blockName])
                    SetPrimaryPartCFrame(clonedBlock, cframe)
                    clonedBlock.Health.Value = ""
                    clonedBlock.Parent = BuildPreview
                    local pPart = clonedBlock.PPart
                    -- Scale size to make blocks smaller
                    pPart.Size = block.Size and stringToVector3(block.Size) * scaleFactor or pPart.Size * scaleFactor
                    pPart.Anchored = true
                    pPart.Transparency = transparency -- Apply mesh transparency
                    pPart.CastShadow = block.ShowShadow or false
                    pPart.CanCollide = block.CanCollide or true
                    -- Apply color if specified
                    if block.Color then
                        local color = Color3new(unpack({block.Color:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")}))
                        for _, descendant in pairs(GetDescendants(clonedBlock)) do
                            if descendant:IsA("BasePart") then
                                descendant.Color = color
                            end
                        end
                    end
                    blockCount = blockCount + 1
                    task.wait()
                end)
            end
        end
    end
    WindUI:Notify({
        Title = blockCount > 0 and "Success" or "No Blocks",
        Content = blockCount > 0 and ("Previewed " .. blockCount .. " refined blocks (Mesh Level: " .. meshLevel .. ", Transparency: " .. transparency .. ")") or "No valid blocks in file!",
        Icon = blockCount > 0 and "check" or "info",
        Duration = 5
    })
end

-- Step 1: Image Loader Functions
local function readFile(filePath)
    if not isfile(filePath) then
        return nil
    end
    return readfile(filePath)
end

local function parseColors(fileContent)
    local data = {}
    for value in string.gmatch(fileContent, "[^,]+") do
        value = value:match("^%s*(.-)%s*$")
        table.insert(data, tonumber(value) or value)
    end
    return data
end

local function calculateFrameSize(data)
    local width = 0
    local height = 0
    local currentWidth = 0
    for i = 1, #data, 3 do
        local r, g, b = data[i], data[i + 1], data[i + 2]
        if r == "B" and g == "B" and b == "B" then
            height = height + 1
            width = math.max(width, currentWidth)
            currentWidth = 0
        elseif r == "R" and g == "R" and b == "R" then
            currentWidth += 1
        elseif type(r) == "number" and type(g) == "number" and type(b) == "number" then
            currentWidth += 1
        end
    end
    height += 1
    width = math.max(width, currentWidth)
    return Vector3.new(width * blockSize, height * blockSize, blockDepth)
end

local function buildImagePreview(data, blockSize, blockDepth, rotation, offset)
    BuildPreview:ClearAllChildren()
    local frameSize = calculateFrameSize(data)
    local plot = Zone
    if not plot then
        WindUI:Notify({
            Title = "Error",
            Content = "Team zone not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local startPosition = plot.Position + Vector3.new(-100, 150, 0) + offset
    local centerCFrame = CFrame.new(startPosition + Vector3.new(frameSize.X / 2, -frameSize.Y / 2, 0))
    local rotationCFrame = CFrame.Angles(0, math.rad(rotation - 90), 0)
    local brainRot = centerCFrame * rotationCFrame
    local currentX = startPosition.X
    local currentY = startPosition.Y
    local initialX = startPosition.X
    local dataIndex = 1
    local batchSize = previewSpeed
    local currentConnection
    currentConnection = game:GetService("RunService").Heartbeat:Connect(function()
        local finished = false
        for _ = 1, batchSize do
            if dataIndex > #data then
                finished = true
                break
            end
            local r, g, b = data[dataIndex], data[dataIndex + 1], data[dataIndex + 2]
            if r == "B" and g == "B" and b == "B" then
                currentX = initialX
                currentY = currentY - blockSize
            elseif r == "R" and g == "R" and b == "R" then
                currentX = currentX + blockSize
            elseif type(r) == "number" and type(g) == "number" and type(b) == "number" then
                local block = Clone(BuildingParts[blockType])
                local pPart = block.PPart
                pPart.Size = Vector3.new(blockSize, blockSize, blockDepth)
                pPart.Color = Color3.fromRGB(r, g, b)
                pPart.Anchored = true
                pPart.CastShadow = false
                pPart.CanCollide = false
                block.Health.Value = ""
                local blockPosition = Vector3.new(currentX, currentY, startPosition.Z) + offset
                local relativeCFrame = centerCFrame:ToObjectSpace(CFrame.new(blockPosition))
                SetPrimaryPartCFrame(block, brainRot * relativeCFrame)
                block.Parent = BuildPreview
                currentX = currentX + blockSize
            end
            dataIndex += 3
        end
        if finished then
            currentConnection:Disconnect()
        end
    end)
end

local function buildImageFast(data, blockSize, blockDepth, rotation, offset)
    local plot = Zone
    if not plot then
        WindUI:Notify({
            Title = "Error",
            Content = "Team zone not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local parts = {}
    for _, child in ipairs(BuildPreview:GetChildren()) do
        if child:IsA("Model") and child.PPart then
            table.insert(parts, child.PPart)
        end
    end
    if #parts == 0 then
        WindUI:Notify({
            Title = "Error",
            Content = "No preview blocks found! Enable preview first.",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end
    local paintData = {}
    local playerName = game.Players.LocalPlayer.Settings.ShareBlocks.Value and game:GetService("Teams")[player.Team.Name].TeamLeader.Value or player.Name
    local blocksFolder = workspace.Blocks:FindFirstChild(playerName)
    local initialBlockCount = blocksFolder and #blocksFolder:GetChildren() or 0
    for i, part in ipairs(parts) do
        if not getgenv().ImgLoaderStat then break end
        task.spawn(function()
            local worldPos = part.Position
            local partRot = part.CFrame - part.Position
            local tempCFrame = CFrame.new(math.random(-69, 69), math.random(-2200000, -120000), math.random(-69, 69))
            local buildTool = player.Backpack:FindFirstChild("BuildingTool") or player.Character:FindFirstChild("BuildingTool")
            buildTool.RF:InvokeServer(blockType, UserBlockList[blockType], Zone, tempCFrame, true)
        end)
    end
    local blocks = blocksFolder and blocksFolder:GetChildren() or {}
    while #blocks < initialBlockCount + #parts and getgenv().ImgLoaderStat do
        task.wait(0.5)
        blocks = blocksFolder and blocksFolder:GetChildren() or {}
    end
    for i, part in ipairs(parts) do
        if not getgenv().ImgLoaderStat then break end
        task.spawn(function()
            local worldPos = part.Position
            local partRot = part.CFrame - part.Position
            local newCFrame = CFrame.new(worldPos) * partRot * CFrame.Angles(0, math.rad(90), 0)
            local targetBlock = blocks[initialBlockCount + i]
            local scaleTool = player.Backpack:FindFirstChild("ScalingTool") or player.Character:FindFirstChild("ScalingTool")
            scaleTool.RF:InvokeServer(targetBlock, Vector3.new(blockDepth, blockSize, blockSize), newCFrame)
            table.insert(paintData, {targetBlock, part.Color})
        end)
    end
    local paintTool = player.Backpack:FindFirstChild("PaintingTool") or player.Character:FindFirstChild("PaintingTool")
    paintTool.RF:InvokeServer(paintData)
    BuildPreview:ClearAllChildren()
    local deleteTool = player.Character:FindFirstChild("DeleteTool") or player.Backpack:FindFirstChild("DeleteTool")
    for _, block in ipairs(blocks) do
        task.spawn(function()
            local blockY = block.PrimaryPart and block.PrimaryPart.Position.Y or nil
            if blockY and blockY >= -2200000 and blockY <= -120000 then
                deleteTool.RF:InvokeServer(block)
            end
        end)
    end
    if getgenv().ImgLoaderStat then
        WindUI:Notify({
            Title = "Success",
            Content = "Image successfully loaded!",
            Icon = "check",
            Duration = 5
        })
    else
        WindUI:Notify({
            Title = "Aborted",
            Content = "Image build aborted. Unplaced blocks deleted.",
            Icon = "triangle-alert",
            Duration = 5
        })
    end
end

-- AutoFarm Function
local function autoFarm(currentRun)
    -- Ensure player character exists
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then return end

    -- Variables
    local Character = LocalPlayer.Character
    local NormalStages = Workspace.BoatStages.NormalStages

    -- Go to each stage
    for i = 1, 10 do
        if not getgenv().TreasureAutoFarm.Enabled then return end

        local Stage = NormalStages["CaveStage" .. i]
        local DarknessPart = Stage and Stage:FindFirstChild("DarknessPart")

        if DarknessPart then
            -- Teleport to next stage
            print("Teleporting to Stage " .. i)
            Character.HumanoidRootPart.CFrame = DarknessPart.CFrame

            -- Create a temp part under you
            local Part = Instance.new("Part", LocalPlayer.Character)
            Part.Anchored = true
            Part.Position = LocalPlayer.Character.HumanoidRootPart.Position - Vector3.new(0, 6, 0)

            -- Wait and remove temp part
            wait(getgenv().TreasureAutoFarm.Teleport)
            Part:Destroy()
        end
    end

    -- Teleport to the end
    print("Teleporting to the end")
    repeat wait()
        Character.HumanoidRootPart.CFrame = NormalStages.TheEnd.GoldenChest.Trigger.CFrame
    until Lighting.ClockTime ~= 14

    -- Wait for respawn
    local Respawned = false
    LocalPlayer.CharacterAdded:Connect(function()
        Respawned = true
    end)

    repeat wait() until Respawned
    wait(getgenv().TreasureAutoFarm.TimeBetweenRuns)
    print("AutoFarm: Run " .. currentRun .. " completed")
end

-- Time Formatting Function
local function formatTime(seconds)
    local hours = math.floor(seconds / 3600)
    local minutes = math.floor((seconds % 3600) / 60)
    local sec = seconds % 60
    return string.format("%02d:%02d:%02d", hours, minutes, sec)
end

-- Update Stats Function
local function updateStats(statsLabel)
    while trackerEnabled do
        local currentGold = player.Data.Gold.Value
        local Ftime = formatTime(clockTime)
        local goldGained = currentGold - lastGoldValue
        totalGoldGained = totalGoldGained + goldGained

        local currentGoldBlock = player.Data.GoldBlock.Value
        totalGoldBlock = currentGoldBlock - IGBLOCK

        GoldPerHour = (clockTime > 0) and ((totalGoldGained / clockTime) * 3600) or 0

        -- Update the Gold Tracker Stats
        statsLabel.Text = "Elapsed Time: " .. Ftime ..
                          "\nGold Gained: " .. totalGoldGained ..
                          "\nGold Blocks Gained: " .. totalGoldBlock ..
                          "\nGold per Hour: " .. math.floor(GoldPerHour)

        lastGoldValue = currentGold
        task.wait(1)
    end
end

-- Start Timer for Gold Tracker
local function startGoldTrackerTimer()
    if running then return end
    running = true
    while running and trackerEnabled do
        clockTime = clockTime + 1
        task.wait(1)
    end
end

-- Apply Fly Speed using BodyVelocity
local function ApplyFlySpeed(speed)
    local humanoidRootPart = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if humanoidRootPart then
        -- Remove any existing BodyVelocity (if present)
        local bodyVelocity = humanoidRootPart:FindFirstChild("BodyVelocity")

        bodyVelocity = Instance.new("BodyVelocity")

local function StartFarming()
    while _G.FarmMoney do

        Teleport(59.148, 65.990, 825.943)
        task.wait(0.5)  

        -- 2️⃣ Set gravity & Apply Flight Speed (Reapply after teleport)

        ApplyFlySpeed(FlySpeed)
        task.wait(5)  

        -- 3️⃣ Teleport to chest location
        Teleport(-55.558, -359.302, 9495.745)
        task.wait(ChestWaitTime)  

        -- 4️⃣ Wait for chest collection
        local TriggerChest = workspace.BoatStages.NormalStages.TheEnd.GoldenChest.Trigger
        if TriggerChest then
            TriggerChest.Touched:Wait()
        end

        -- 5️⃣ Teleport to **adjustable position**
        Teleport(TeleportLeftRight, TeleportUpDown, TeleportForward)
        SetGravity(GravityValue)  -- Set gravity again in case it resets
        ApplyFlySpeed(FlySpeed) -- Reapply fly speed
        task.wait(0.2)  -- Stabilize position

        -- 6️⃣ **Teleport to the final position 3489**
        Teleport(-34.82, 63.44, 3489.43)  -- This is the teleport you requested
        SetGravity(GravityValue)  -- Set gravity
        ApplyFlySpeed(FlySpeed) -- Apply fly speed again after teleport
        task.wait(0.5) -- Stabilize position

        -- 7️⃣ Wait before restarting the farm
        task.wait(AutoFarmRestartTime)
    end
end

local function FarmingLoop()
    while _G.FarmMoney do

        end)

        Teleport(defaultTeleportPosition.X, defaultTeleportPosition.Y, defaultTeleportPosition.Z)
        
        -- Reset velocity & stabilize position
        ResetVelocity()
        task.wait() -- Wait a bit before movement starts

        -- Set gravity to 1 and apply fly speed for forward movement
        SetGravity(gravityValue)
        ApplyFlySpeed(flySpeed)

        -- Wait 19 seconds, then force the player to die
        task.wait(19)
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
            LocalPlayer.Character.Humanoid.Health = 0 -- Kill the player
        end

        -- Wait for the player to respawn before restarting
        LocalPlayer.CharacterAdded:Wait()
        task.wait(1) -- Give a small delay before restarting
    end
end

local function stringToVector3(str)
    local x, y, z = str:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")
    return Vector3.new(tonumber(x), tonumber(y), tonumber(z))
end

local function stringToRotationCFrame(str)
    local rx, ry, rz = str:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")
    return CFrame.Angles(math.rad(tonumber(rx)), math.rad(tonumber(ry)), math.rad(tonumber(rz)))
end

local function mimicBuild(filePath)
    if not filePath or not isfile(filePath) then
        WindUI:Notify({
            Title = "Error",
            Content = "No file selected or file does not exist!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end

    local success, content = pcall(readfile, filePath)
    if not success then
        WindUI:Notify({
            Title = "Error",
            Content = "Failed to read file: " .. tostring(content),
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end

    local jsonData
    success, jsonData = pcall(HttpService.JSONDecode, HttpService, content)
    if not success then
        WindUI:Notify({
            Title = "Error",
            Content = "Invalid JSON data in file!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end

    local teamZone = getTeamZone(player.Team.Name)
    if not teamZone then
        WindUI:Notify({
            Title = "Error",
            Content = "Player's team zone not found!",
            Icon = "triangle-alert",
            Duration = 5
        })
        return
    end

    local teamCFrame = teamZone.CFrame

    for _, child in ipairs(BuildMimicPreview:GetChildren()) do
        child:Destroy()
    end

    local blockCount = 0

    for blockName, blockList in pairs(jsonData) do
        for _, blockInfo in ipairs(blockList) do
            local model = Instance.new("Model")
            model.Name = blockName
            model.Parent = BuildMimicPreview

            local pPart = Instance.new("Part")
            pPart.Name = "PPart"
            pPart.Anchored = true
            pPart.CanCollide = blockInfo.CanCollide or false
            pPart.CastShadow = blockInfo.ShowShadow or false
            pPart.Transparency = blockInfo.Transparency or 0
            pPart.Size = blockInfo.Size and stringToVector3(blockInfo.Size) or Vector3.new(1, 1, 1)
            pPart.Parent = model

            local position = stringToVector3(blockInfo.Position)
            local rotation = stringToRotationCFrame(blockInfo.Rotation)
            pPart.CFrame = teamCFrame * CFrame.new(position) * rotation

            if blockInfo.Color then
                local r, g, b = blockInfo.Color:match("([%d.-]+),%s*([%d.-]+),%s*([%d.-]+)")
                pPart.Color = Color3.new(tonumber(r), tonumber(g), tonumber(b))
            elseif NormalColorBlock[blockName] then
                pPart.Color = Color3.new(unpack(NormalColorBlock[blockName]))
            end

            if blockInfo.Transparency and blockInfo.Transparency > 0 then
                local transparencyModifier = Instance.new("NumberValue")
                transparencyModifier.Name = "TransparencyModifier"
                transparencyModifier.Value = blockInfo.Transparency
                transparencyModifier.Parent = model
            end

            blockCount += 1
        end
    end

    WindUI:Notify({
        Title = blockCount > 0 and "Success" or "No Blocks",
        Content = blockCount > 0 and ("Mimicked " .. blockCount .. " blocks") or "No blocks found in the file!",
        Icon = blockCount > 0 and "check" or "info",
        Duration = 5
    })
end
